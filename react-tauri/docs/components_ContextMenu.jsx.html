import { useEffect } from "react";

/**
  * @name ContextMenu
  * @description
  * ----
  * 
  * ###### Description
  * - ContextMenu is a reusable component that displays a menu at a specific location on the screen.
  * 
  * - ContextMenu is used when users interact with elements that require additional actions, such as right-clicking or selecting a menu button.
  * 
  * - ContextMenu receives its position through the *x* and *y* props and displays its children inside the menu container.
  * 
  * - ContextMenu automatically closes when the user clicks outside the menu or presses the Escape key.
  * 
  * ----
  * 
  * 
  * ###### Impelentation Logic
  * 
  * ContextMenu receives the following props:
  * 
  * ```visible```
  * 
  * - Controls whether the menu is displayed or hidden.
  * 
  * ```onClose```
  * 
  * - A function passed from the parent component used to close the menu.
  * 
  * ```children```
  * 
  * - The content that will be displayed inside the menu.
  * 
  * 
  * **useEffect()**
  * 
  * - useEffect watches the *visible* state of the menu.
  * 
  * - When ContextMenu becomes visible, it adds event listeners to detect outside clicks and Escape key presses.
  * 
  * - When either event occurs, it calls the *onClose()* function.
  * 
  * - When ContextMenu becomes hidden or is removed, the event listeners are cleaned up.
  * 
  * 
  * **handleOutsideClick()**
  * 
  * - handleOutsideClick is a helper function.
  * 
  * - Closes the menu when the user clicks anywhere outside of ContextMenu.
  * 
  * 
  * **handleEscape()**
  * 
  * - handleEscape is a helper function.
  * 
  * - Checks if the user presses the Escape key.
  * 
  * - Closes the menu if the Escape key is pressed.
  * 
  * 
  * The menu only renders when *visible* is true.
  * 
  * If *visible* is false, ContextMenu returns null and does not display anything.
  * 
  * 
  * The menu position is controlled using inline styles:
  * 
  * ```style={{
  *   left: x,
  *   top: y,
  * }}
  * ```
  * 
  * This allows the menu to appear where the user interacted with the screen.
  * 
  * 
  * The menu uses *stopPropagation()* to prevent clicks inside the menu from triggering the outside click listener.
  * 
  * ----
  * 
  * @param {number} x - The horizontal position of the context menu
  * @param {number} y - The vertical position of the context menu
  * @param {boolean} visible - Determines if the context menu is displayed
  * @param {Function} onClose - Function used to close the context menu
  * @param {ReactNode} children - The content displayed inside the context menu
  */

export default function ContextMenu({
  x,
  y,
  visible,
  onClose,
  children,
}) {
  useEffect(() => {
    if (!visible) return;

    function handleOutsideClick() {
      onClose();
    }

    function handleEscape(event) {
      if (event.key === "Escape") {
        onClose();
      }
    }

    window.addEventListener("click", handleOutsideClick);
    window.addEventListener("keydown", handleEscape);

    return () => {
      window.removeEventListener("click", handleOutsideClick);
      window.removeEventListener("keydown", handleEscape);
    };
  }, [visible, onClose]);

  if (!visible) return null;

  return (
    <div
      className="fixed border bg-(--bg) border-white rounded-lg py-2 min-w-52 z-50"
      style={{
        left: x,
        top: y,
      }}
      onClick={(e) => e.stopPropagation()}
    >
      {children}
    </div>
  );
}
